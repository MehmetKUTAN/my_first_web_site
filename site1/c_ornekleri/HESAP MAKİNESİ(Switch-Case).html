HESAP MAKİNESİ(Switch-Case)

#include <stdio.h>
#include <stdlib.h>

int main()
{
    int sayi_1,sayi_2;
    char opr;
    float sonuc;
    printf("1.sayiyi giriniz=\n");
    scanf("%d",&sayi_1);
    printf("2.sayiyi giriniz=\n");
    scanf("%d",&sayi_2);
    printf("hangi işlemi yapmak istiyorsunuz(+,-,*,/)=");
    scanf(" %c",&opr);
    switch(opr)
    {
    case '+':
        sonuc=sayi_1+sayi_2;
        break;

        case '-':
        sonuc=sayi_1-sayi_2;
        break;

        case '*':
        sonuc=sayi_1*sayi_2;
        break;

        case '/':
        sonuc=sayi_1/sayi_2;
        break;

        default:
            printf("hatalı oparatör tikladiniz");

    }

    printf("işlem sonucu=%3f",sonuc);
    return 0;
}
