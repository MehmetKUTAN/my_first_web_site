Girilen sayının tersini yazma

#include <stdio.h>
#include <stdlib.h>

int main()
{


    int sayi,a, sonuc=0, kalan;

    printf("Bİr Sayi Giriniz: ");
    scanf("%d", &sayi);
    a=sayi;
    while(a!=0){
        kalan=a%10;
        sonuc=sonuc*10+kalan;
        a/=10;
    }
    printf("%d sayının tersi=%d",sayi,sonuc);
    return 0;
}
