Kullanıcının İstediği kadar sayı ortalamasını bulan C programı(FOR döngüsü)

#include <stdio.h>
#include <stdlib.h>

int main()
{
int sayi_miktari,i;
float sayi,ort,toplam=0;
printf("kac adet sayi gireceksiniz=");
scanf("%d",&sayi_miktari);
for(i=1;i<=sayi_miktari;i++)
{
printf("%d.sayiyi giriniz=",i);
scanf("%f",&sayi);
toplam+=sayi;

}
i-=1;
ort=toplam/(sayi_miktari*1.0);
printf("%d tane sayının ortalamasi=%.2f",i,ort);
return 0;
}
