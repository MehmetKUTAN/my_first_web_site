En büyük ve En kücük rakamı bulan 
C programı


#include <stdio.h>
#include <stdlib.h>

int main()
{
    int sayi,enb=0,enk=9,rakam;
    printf("bir sayi giriniz=");
    scanf("%d",&sayi);
    rakam= sayi % 10;
   while(sayi>1)
    {
        rakam= sayi % 10;
        if(rakam>enb)
            enb=rakam;
        if(rakam<enk)
            enk=rakam;
    sayi=sayi/10;

    }
        printf("en buyuk rakam=%d \n",enb);
        printf("en kucuk rakam=%d \n",enk);
    return 0;
}
