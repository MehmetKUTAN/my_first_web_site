Mükemmel sayıyı bulan C programı

#include <stdio.h>
#include <stdlib.h>

int main()
{
    int sayi,i,top=0;
    printf("Bir sayi giriniz=");
    scanf("%d",&sayi);
    for(i=1;i<sayi;i++)
    {
        if(sayi%i==0)
        {
            top=top+i;
        }
    }
    if(top==sayi)
    {
        printf("mukkemel sayi");
    }
    else
        printf("mükemmel sayi degildir");
    return 0;
}
