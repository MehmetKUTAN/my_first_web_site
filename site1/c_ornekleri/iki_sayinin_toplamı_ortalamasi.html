İki sayının kullanıcıdan alınması ve ortalamasını bulan c programı.

#include <stdio.h>
#include <stdlib.h>

int main()
{
    int sayi_1,sayi_2,ort;
    printf("1,sayiyi giriniz=");
    scanf("%d",&sayi_1);
    printf("2.sayiyi girniz=");
    scanf("%d",&sayi_2);
    ort=(sayi_1 + sayi_2)/2;
    printf("2 sayinin ortalamasý=%d",ort);
    return 0;
}
