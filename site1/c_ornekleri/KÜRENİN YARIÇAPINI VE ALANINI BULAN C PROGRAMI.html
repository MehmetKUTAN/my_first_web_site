KÜRENİN YARIÇAPINI VE ALANINI BULAN C PROGRAMI

 


#include <stdio.h>
#include <stdlib.h>
int main()
{
float pi=3.14,yaricap,alan,hacim;
printf("kürenin yariçapini giriniz=");                     /*kürenin yarýçapýný ister*/
scanf("%f",&yaricap);
alan= 4*pi*yaricap*yaricap;                               /*alan förmülü*/
hacim=4/3*pi*yaricap*yaricap*yaricap;                    /*cevre formülü*/
printf("kürenin alani=%.2f\n",alan);                    /*kürenin alaný*/
printf("kürenin hacmi=%.3f\n",hacim);                  /*kürenin hacmi*/
return 0;
}
